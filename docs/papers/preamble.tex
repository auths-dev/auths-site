% LaTeX preamble injected into every paper here via pandoc --include-in-header.
% Design system: STIX Two (the OpenType family designed for scientific
% publishing) + unicode-math, compact titlesec headings, styled captions,
% running heads, tightened lists. Kept separate from the Markdown so the
% source stays clean and portable.

% ---- math (amssymb/amsfonts deliberately NOT loaded: unicode-math replaces them)
\usepackage{amsmath}
\usepackage{booktabs}
\usepackage{fancyvrb}
\usepackage{float}                       % [H]: pin a figure exactly in place
\usepackage{microtype}

% ---- fonts: STIX Two via fontspec/unicode-math, guarded fallbacks ----------
\usepackage{iftex}
\ifXeTeX\usepackage{fontspec}\usepackage{unicode-math}\fi
\ifLuaTeX\usepackage{fontspec}\usepackage{unicode-math}\fi
\makeatletter
\@ifpackageloaded{fontspec}{%
  \IfFontExistsTF{STIXTwoText-Regular.otf}{%
    \setmainfont{STIXTwoText}[
      Extension=.otf, UprightFont=*-Regular, BoldFont=*-Bold,
      ItalicFont=*-Italic, BoldItalicFont=*-BoldItalic]%
  }{}%
  \@ifpackageloaded{unicode-math}{%
    \IfFontExistsTF{STIXTwoMath-Regular.otf}{%
      \setmathfont{STIXTwoMath-Regular.otf}%
    }{}%
  }{}%
  \IfFontExistsTF{texgyreheros-regular.otf}{%
    \setsansfont{texgyreheros}[
      Extension=.otf, UprightFont=*-regular, BoldFont=*-bold,
      ItalicFont=*-italic, BoldItalicFont=*-bolditalic,
      Scale=MatchLowercase]%
  }{}%
  \IfFontExistsTF{Menlo}{\setmonofont{Menlo}[Scale=0.78]}{%
    \IfFontExistsTF{Inconsolatazi4-Regular.otf}{%
      \setmonofont{Inconsolatazi4}[Extension=.otf,UprightFont=*-Regular,BoldFont=*-Bold,Scale=MatchLowercase]}{}}%
}{}
\makeatother
\linespread{1.04}
\frenchspacing

% ---- theorem-like environments (definitions, propositions) -----------------
\usepackage{amsthm}
\newtheorem{definition}{Definition}
\newtheorem{proposition}{Proposition}
\newtheorem{theorem}{Theorem}

% ---- headings: compact, with slate-colored numbers -------------------------
\usepackage{xcolor}                      % no-op if pandoc already loaded it
\usepackage{titlesec}
\definecolor{papernum}{RGB}{60,76,102}   % slate for section numbers + links
\titleformat{\section}
  {\large\bfseries}{\textcolor{papernum}{\thesection}}{0.9em}{}
\titleformat{\subsection}
  {\normalsize\bfseries}{\textcolor{papernum}{\thesubsection}}{0.8em}{}
\titlespacing*{\section}{0pt}{*2.6}{*1.1}
\titlespacing*{\subsection}{0pt}{*1.8}{*0.7}

% ---- captions, lists, code frames ------------------------------------------
\usepackage[font=small,labelfont=bf,skip=5pt]{caption}
\usepackage{enumitem}
\setlist{itemsep=1.5pt,topsep=3pt,parsep=0pt}
\fvset{rulecolor=\color{black!40}}       % lighter frames on Verbatim boxes

% ---- running heads ----------------------------------------------------------
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\footnotesize\itshape Accountability without consensus, freshness with a witness}
\fancyhead[R]{\footnotesize\thepage}
\renewcommand{\headrulewidth}{0.2pt}
\fancypagestyle{plain}{\fancyhf{}\fancyfoot[C]{\footnotesize\thepage}%
  \renewcommand{\headrulewidth}{0pt}}

% ---- links -------------------------------------------------------------------
\AtBeginDocument{\hypersetup{linkcolor=papernum,citecolor=papernum,urlcolor=papernum}}

% ---- TikZ (architecture diagrams) -------------------------------------------
\usepackage{tikz}
\usetikzlibrary{arrows.meta,positioning,shapes.geometric,fit,backgrounds,calc}
\tikzset{
  box/.style={draw,rounded corners=2pt,align=center,inner sep=5pt,font=\small},
  lbl/.style={font=\scriptsize\itshape},
  gate/.style={draw,fill=black!5,rounded corners=2pt,align=center,inner sep=6pt,font=\small\bfseries}
}

% ---- figure design system ----------------------------------------------------
% Pastel ownership zones, dashed color-coded component boxes with icons,
% arrows inheriting their component's accent, all-sans figure text.
\usepackage{fontawesome5}
\usepackage[skins]{tcolorbox}
\definecolor{gblue}{RGB}{66,133,244}
\definecolor{gred}{RGB}{217,48,37}
\definecolor{gamber}{RGB}{242,153,0}
\definecolor{ggreen}{RGB}{52,168,83}
\definecolor{gpurple}{RGB}{124,77,255}
\definecolor{zonegreen}{RGB}{230,244,234}
\definecolor{zoneblue}{RGB}{232,240,254}
\definecolor{zoneamber}{RGB}{254,247,224}
\definecolor{ink}{RGB}{60,64,67}
\tikzset{
  comp/.style={draw=#1, dashed, line width=0.9pt, rounded corners=3pt,
    fill=white, align=center, inner xsep=7pt, inner ysep=5pt,
    font=\sffamily\footnotesize},
  compsolid/.style={draw=#1, line width=1pt, rounded corners=3pt,
    fill=white, align=center, inner xsep=8pt, inner ysep=6pt,
    font=\sffamily\footnotesize},
  chip/.style={draw=#1, line width=0.8pt, rounded corners=6pt, fill=white,
    font=\sffamily\scriptsize\bfseries, text=#1, inner xsep=7pt, inner ysep=3.5pt},
  zone/.style={fill=#1, rounded corners=6pt, draw=none},
  zlabel/.style={font=\sffamily\bfseries\small, text=ink, align=center},
  elabel/.style={font=\sffamily\scriptsize, text=ink, align=center, fill=none},
  flow/.style={->, >=Stealth, line width=1pt, draw=#1, shorten >=2pt, shorten <=2pt},
  flow/.default=ink,
}
% Code panel with a colored dashed border and colored icon+title, and the
% pastel zone container the panels sit on.
\newtcolorbox{panelbox}[3][]{enhanced, colback=white, frame hidden, boxrule=0pt,
  borderline={0.9pt}{0pt}{#2, dashed}, arc=3pt,
  left=6pt, right=6pt, top=2pt, bottom=3pt,
  title={\sffamily\bfseries\footnotesize\textcolor{#2}{#3}},
  colbacktitle=white, coltitle=#2, titlerule=0pt, halign title=center,
  toptitle=4pt, bottomtitle=1pt, #1}
\newtcolorbox{zonepanel}[2][]{enhanced, colback=#2, frame hidden, boxrule=0pt,
  arc=6pt, left=8pt, right=8pt, top=7pt, bottom=8pt, #1}
